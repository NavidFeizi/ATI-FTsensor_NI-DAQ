set(ATIFT_LIB_SOURCES
    src/ATI_FT.cpp
    libs/dom.c
    libs/expatls.c
    libs/ftconfig.c
    libs/ftrt.c
    libs/node.c
    libs/stack.c
    libs/xmlparse.c
    libs/xmlrole.c
    libs/xmltok.c
    )

			PATHS ${DAQmx_DIR}

target_link_libraries(ATIlib ${NIDAQmx})
